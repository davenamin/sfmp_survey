  \begin{tabular}{@{}p{#1}@{}}
    #2 \\[2\normalbaselineskip] \hrule \\[0pt]
    {\small \textit{Name}} \\[2\normalbaselineskip] \hrule
    \\[0pt]
    {\small \textit{Signature}} \\[2\normalbaselineskip] \hrule \\[0pt]

    {\small \textit{Date}}
  \end{tabular}
}
